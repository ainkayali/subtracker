# Fastmail Mail-Based Subscription Detection — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans. Steps use checkbox (`- [ ]`) syntax.

**Goal:** SubTracker'a Fastmail mail tabanlı abonelik tespit (real-time push + 12 ay backfill) ekle; OCI sunucusunda `subhook` Node.js servisi mailleri DeepSeek ile parse edip FCM push'la Android cihaza iletir.

**Architecture:** Iki bileşen — `subhook` Express server (OCI, systemd user) ve SubTracker Android (yeni FCM + PendingDetection katmanı). Server JMAP push'larını alır, mailleri DeepSeek'e gönderir, FCM data message ile cihazı bilgilendirir. Cihaz local DB ile match yapar, kullanıcı her tespiti onaylar.

**Tech Stack:**
- **Server:** Node.js 20+, Express, better-sqlite3, ofetch, firebase-admin, zod, jose (HMAC)
- **Android:** Kotlin/Compose (mevcut), Firebase Cloud Messaging (BoM 32.7.x), OkHttp, kotlinx.serialization, Room v3 migration
- **Infra:** OCI host (mevcut), cloudflared (mevcut), user-systemd (`linger=yes`)

**Spec referansı:** `docs/superpowers/specs/2026-05-11-fastmail-detect-design.md`

// src/db.js
import Database from 'better-sqlite3';

const SCHEMA = [
  `CREATE TABLE IF NOT EXISTS jmap_state (
    id INTEGER PRIMARY KEY CHECK (id = 1),
    state TEXT NOT NULL,
    updated_at INTEGER NOT NULL
  )`,
  `CREATE TABLE IF NOT EXISTS fcm_devices (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    token TEXT NOT NULL UNIQUE,
    device_name TEXT,
    registered_at INTEGER NOT NULL
  )`,
  `CREATE TABLE IF NOT EXISTS processed_emails (
    email_id TEXT PRIMARY KEY,
    processed_at INTEGER NOT NULL,
    filtered_out INTEGER NOT NULL,
    is_subscription INTEGER,
    detection_json TEXT
  )`,
  `CREATE INDEX IF NOT EXISTS idx_processed_at ON processed_emails(processed_at)`,
  `CREATE TABLE IF NOT EXISTS llm_call_log (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    called_at INTEGER NOT NULL,
    email_id TEXT,
    duration_ms INTEGER,
    ok INTEGER
  )`,
  `CREATE INDEX IF NOT EXISTS idx_llm_called_at ON llm_call_log(called_at)`
];

export function openDb(path) {
  const db = new Database(path);
  db.pragma('journal_mode = WAL');
  db.pragma('foreign_keys = ON');
  for (const stmt of SCHEMA) db.prepare(stmt).run();
  return db;
}

export function getJmapState(db) {
  return db.prepare('SELECT state FROM jmap_state WHERE id=1').get()?.state ?? null;
}

export function setJmapState(db, state) {
  db.prepare(`INSERT INTO jmap_state (id, state, updated_at) VALUES (1, ?, ?)
              ON CONFLICT(id) DO UPDATE SET state=excluded.state, updated_at=excluded.updated_at`)
    .run(state, Date.now());
}

export function markEmailProcessed(db, { emailId, filteredOut, isSubscription, detectionJson }) {
  db.prepare(`INSERT INTO processed_emails (email_id, processed_at, filtered_out, is_subscription, detection_json)
              VALUES (?, ?, ?, ?, ?)
              ON CONFLICT(email_id) DO UPDATE SET
                processed_at = excluded.processed_at,
                filtered_out = excluded.filtered_out,
                is_subscription = excluded.is_subscription,
                detection_json = excluded.detection_json`)
    .run(emailId, Date.now(), filteredOut ? 1 : 0, isSubscription == null ? null : (isSubscription ? 1 : 0), detectionJson ?? null);
}

export function isEmailProcessed(db, emailId) {
  return !!db.prepare('SELECT 1 FROM processed_emails WHERE email_id=?').get(emailId);
}

export function registerFcmToken(db, token, deviceName) {
  db.prepare(`INSERT INTO fcm_devices (token, device_name, registered_at) VALUES (?, ?, ?)
              ON CONFLICT(token) DO UPDATE SET registered_at=excluded.registered_at`)
    .run(token, deviceName ?? null, Date.now());
}

export function listFcmTokens(db) {
  return db.prepare('SELECT token FROM fcm_devices').all().map(r => r.token);
}

export function llmCallsInLastDay(db) {
  const since = Date.now() - 86_400_000;
  return db.prepare('SELECT COUNT(*) as c FROM llm_call_log WHERE called_at >= ?').get(since).c;
}

export function logLlmCall(db, { emailId, durationMs, ok }) {
  db.prepare('INSERT INTO llm_call_log (called_at, email_id, duration_ms, ok) VALUES (?, ?, ?, ?)')
    .run(Date.now(), emailId ?? null, durationMs, ok ? 1 : 0);
}
```

// src/auth.js
import { createHmac, timingSafeEqual } from 'node:crypto';

const MAX_DRIFT_MS = 5 * 60_000;

export function signRequest({ secret, ts, body }) {
  const mac = createHmac('sha256', secret).update(`${ts}.${body}`).digest('base64');
  return `HMAC-SHA256 ts=${ts},sig=${mac}`;
}

export function verifyRequest({ secret, header, body, now = Date.now() }) {
  if (!header) return { ok: false, reason: 'missing header' };
  const m = /^HMAC-SHA256\s+ts=(\d+),sig=([A-Za-z0-9+/=]+)$/.exec(header);
  if (!m) return { ok: false, reason: 'malformed header' };
  const ts = Number(m[1]);
  const sig = m[2];
  if (Math.abs(now - ts) > MAX_DRIFT_MS) return { ok: false, reason: `drift ${now - ts}ms` };
  const expected = createHmac('sha256', secret).update(`${ts}.${body}`).digest('base64');
  const a = Buffer.from(sig);
  const b = Buffer.from(expected);
  if (a.length !== b.length || !timingSafeEqual(a, b)) return { ok: false, reason: 'signature mismatch' };
  return { ok: true };
}

export function hmacMiddleware(secret) {
  return (req, res, next) => {
    let raw = '';
    req.setEncoding('utf8');
    req.on('data', c => { raw += c; });
    req.on('end', () => {
      const result = verifyRequest({ secret, header: req.headers['authorization'], body: raw });
      if (!result.ok) return res.status(401).json({ error: result.reason });
      try {
        req.body = raw ? JSON.parse(raw) : {};
      } catch {
        return res.status(400).json({ error: 'invalid json' });
      }
      next();
    });
  };
}
```

// src/rules.js
import { readFileSync } from 'node:fs';
import { fileURLToPath } from 'node:url';
import { dirname, join } from 'node:path';

const __dirname = dirname(fileURLToPath(import.meta.url));

export function loadRules() {
  const raw = readFileSync(join(__dirname, '..', 'rules.json'), 'utf8');
  const obj = JSON.parse(raw);
  return {
    domains: obj.domains.map(d => d.toLowerCase()),
    keywordRegex: new RegExp(obj.subject_keywords.map(k => k.replace(/[.*+?^${}()|[\]\\]/g, '\\$&')).join('|'), 'i')
  };
}

export function passesPreFilter(rules, { from, subject }) {
  const f = (from ?? '').toLowerCase();
  const s = (subject ?? '');
  if (rules.domains.some(d => f.endsWith('@' + d) || f.endsWith('.' + d))) return true;
  if (rules.keywordRegex.test(s)) return true;
  return false;
}

// src/deepseek.js
import { ofetch } from 'ofetch';
import { DetectionSchema } from '../schemas/detection.js';

const SYSTEM_PROMPT = `Bir abonelik/fatura mailini ayrıştırıyorsun. Yanıt: SADECE geçerli JSON.

Şema:
{
  "is_subscription_payment": boolean,
  "provider": string|null,
  "amount": number|null,
  "currency": string|null,
  "date_iso": "YYYY-MM-DD"|null,
  "cycle": "monthly"|"yearly"|"weekly"|"one-time"|"unknown"|null,
  "confidence": number (0.0-1.0)
}

is_subscription_payment=false ise diğer alanlar null olabilir.
Tutar/tarih bulunamazsa null ver, 0 verme.
Türkçe ve İngilizce mailleri anlarsın.`;

export function buildUserPrompt({ from, subject, body }) {
  return `From: ${from}\nSubject: ${subject}\nBody (first chars):\n${body}`;
}

export function parseDeepSeekResponse(raw) {
  let txt = raw.trim();
  // Strip ```json fences
  const fenced = /^```(?:json)?\s*([\s\S]*?)\s*```$/m.exec(txt);
  if (fenced) txt = fenced[1].trim();
  const parsed = JSON.parse(txt);
  return DetectionSchema.parse(parsed);
}

export async function callDeepSeek({ config, from, subject, body }) {
  const truncated = body.slice(0, config.deepseek.maxBodyChars);
  const userMsg = buildUserPrompt({ from, subject, body: truncated });
  const res = await ofetch(config.deepseek.apiUrl, {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${config.deepseek.apiKey}`,
      'Content-Type': 'application/json'
    },
    body: {
      model: config.deepseek.model,
      messages: [
        { role: 'system', content: SYSTEM_PROMPT },
        { role: 'user', content: userMsg }
      ],
      temperature: 0,
      response_format: { type: 'json_object' }
    },
    timeout: 30_000
  });
  const content = res.choices?.[0]?.message?.content;
  if (!content) throw new Error('DeepSeek empty response');
  return parseDeepSeekResponse(content);
}
```

// src/jmap.js
import { ofetch } from 'ofetch';

// Cache session per-process
let sessionCache = null;

async function getSession(config) {
  if (sessionCache) return sessionCache;
  const session = await ofetch(config.fastmail.sessionUrl, {
    headers: { Authorization: `Bearer ${config.fastmail.token}` }
  });
  sessionCache = {
    accountId: Object.keys(session.accounts)[0],
    apiUrl: session.apiUrl ?? config.fastmail.jmapUrl,
    accounts: session.accounts
  };
  return sessionCache;
}

async function call(config, methodCalls) {
  const session = await getSession(config);
  const body = {
    using: ['urn:ietf:params:jmap:core', 'urn:ietf:params:jmap:mail'],
    methodCalls
  };
  return ofetch(session.apiUrl, {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${config.fastmail.token}`,
      'Content-Type': 'application/json'
    },
    body
  });
}

export async function emailChanges(config, sinceState) {
  const session = await getSession(config);
  const res = await call(config, [
    ['Email/changes', { accountId: session.accountId, sinceState, maxChanges: 100 }, 'c0']
  ]);
  const r = res.methodResponses[0][1];
  return { created: r.created, updated: r.updated, removed: r.removed, newState: r.newState };
}

export async function emailGet(config, ids) {
  if (ids.length === 0) return [];
  const session = await getSession(config);
  const res = await call(config, [
    ['Email/get', {
      accountId: session.accountId,
      ids,
      properties: ['id', 'from', 'subject', 'receivedAt', 'bodyValues', 'textBody', 'preview'],
      fetchTextBodyValues: true
    }, 'c0']
  ]);
  return res.methodResponses[0][1].list;
}

export async function inboxQuery(config, { sinceMillis, limit = 200 }) {
  const session = await getSession(config);
  const res = await call(config, [
    ['Mailbox/query', { accountId: session.accountId, filter: { role: 'inbox' } }, 'm0'],
    ['Email/query', {
      accountId: session.accountId,
      filter: { inMailbox: { resultOf: 'm0', name: 'Mailbox/query', path: '/ids/0' }, after: new Date(sinceMillis).toISOString() },
      sort: [{ property: 'receivedAt', isAscending: false }],
      limit
    }, 'c0']
  ]);
  return res.methodResponses[1][1].ids;
}

export async function currentEmailState(config) {
  const session = await getSession(config);
  const res = await call(config, [
    ['Email/get', { accountId: session.accountId, ids: [], properties: ['id'] }, 'c0']
  ]);
  return res.methodResponses[0][1].state;
}

export function extractPlainText(email) {
  const textBody = email.textBody?.[0];
  if (!textBody) return email.preview ?? '';
  const part = email.bodyValues?.[textBody.partId];
  return part?.value ?? email.preview ?? '';
}

export function extractFromAddress(email) {
  return email.from?.[0]?.email ?? '';
}
```

// src/fcm.js
import { initializeApp, cert } from 'firebase-admin/app';
import { getMessaging } from 'firebase-admin/messaging';
import { readFileSync } from 'node:fs';

let messagingCache = null;

function getMessagingClient(config) {
  if (messagingCache) return messagingCache;
  const sa = JSON.parse(readFileSync(config.fcm.serviceAccountJson, 'utf8'));
  const app = initializeApp({ credential: cert(sa), projectId: config.fcm.projectId });
  messagingCache = getMessaging(app);
  return messagingCache;
}

export async function sendDetectionPush(config, tokens, payload) {
  if (tokens.length === 0) return { successCount: 0 };
  const messaging = getMessagingClient(config);
  const data = {
    type: 'detection',
    email_id: payload.email_id,
    provider: payload.provider ?? '',
    amount: String(payload.amount ?? ''),
    currency: payload.currency ?? '',
    date_iso: payload.date_iso ?? '',
    cycle: payload.cycle ?? '',
    confidence: String(payload.confidence ?? 0),
    raw_subject: (payload.raw_subject ?? '').slice(0, 200)
  };
  return messaging.sendEachForMulticast({
    tokens,
    data,
    android: { priority: 'high' }
  });
}

// src/routes/jmapPush.js
import { Router } from 'express';
import { emailChanges, emailGet, extractPlainText, extractFromAddress } from '../jmap.js';
import { passesPreFilter } from '../rules.js';
import { callDeepSeek } from '../deepseek.js';
import { sendDetectionPush } from '../fcm.js';
import {
  getJmapState, setJmapState, markEmailProcessed,
  listFcmTokens, llmCallsInLastDay, logLlmCall
} from '../db.js';

export function jmapPushRoutes({ config, db, rules }) {
  const r = Router();

  r.post('/jmap-push', async (req, res) => {
    // Fastmail JMAP push posts the changed types; we don't verify signature here (could add IP allowlist)
    res.json({ ok: true }); // ack immediately, process async
    setImmediate(() => processChanges({ config, db, rules }).catch(err =>
      console.error('processChanges error', err)
    ));
  });

  return r;
}

export async function processChanges({ config, db, rules }) {
  let sinceState = getJmapState(db);
  if (!sinceState) {
    const { default: { currentEmailState } } = { default: await import('../jmap.js') };
    sinceState = await currentEmailState(config);
    setJmapState(db, sinceState);
    return;
  }
  const { created, newState } = await emailChanges(config, sinceState);
  setJmapState(db, newState);
  if (created.length === 0) return;
  const emails = await emailGet(config, created);
  const tokens = listFcmTokens(db);

  for (const email of emails) {
    const from = extractFromAddress(email);
    const subject = email.subject ?? '';
    const passes = passesPreFilter(rules, { from, subject });
    if (!passes) {
      markEmailProcessed(db, { emailId: email.id, filteredOut: true });
      continue;
    }
    if (llmCallsInLastDay(db) >= config.deepseek.dailyCap) {
      console.warn('daily LLM cap reached');
      markEmailProcessed(db, { emailId: email.id, filteredOut: false, isSubscription: null });
      continue;
    }
    const body = extractPlainText(email);
    const t0 = Date.now();
    let detection = null;
    let ok = false;
    try {
      detection = await callDeepSeek({ config, from, subject, body });
      ok = true;
    } catch (e) {
      console.error('deepseek error for', email.id, e.message);
    }
    logLlmCall(db, { emailId: email.id, durationMs: Date.now() - t0, ok });
    if (!detection) {
      markEmailProcessed(db, { emailId: email.id, filteredOut: false, isSubscription: null });
      continue;
    }
    markEmailProcessed(db, {
      emailId: email.id,
      filteredOut: false,
      isSubscription: detection.is_subscription_payment,
      detectionJson: JSON.stringify(detection)
    });
    if (!detection.is_subscription_payment || detection.confidence < 0.6) continue;
    await sendDetectionPush(config, tokens, {
      email_id: email.id,
      ...detection,
      raw_subject: subject
    });
  }
}

// src/routes/backfill.js
import { Router } from 'express';
import { hmacMiddleware } from '../auth.js';
import { inboxQuery, emailGet, extractFromAddress, extractPlainText } from '../jmap.js';
import { passesPreFilter } from '../rules.js';
import { callDeepSeek } from '../deepseek.js';
import { isEmailProcessed, markEmailProcessed, logLlmCall, llmCallsInLastDay } from '../db.js';

export function backfillRoutes({ config, db, rules }) {
  const r = Router();
  r.post('/backfill', hmacMiddleware(config.auth.hmacSecret), async (req, res) => {
    const months = Math.max(1, Math.min(24, Number(req.body?.months ?? 12)));
    const sinceMillis = Date.now() - months * 30 * 86_400_000;
    const ids = await inboxQuery(config, { sinceMillis, limit: 500 });
    const detections = [];

    for (const id of ids) {
      if (isEmailProcessed(db, id)) continue;
      const [email] = await emailGet(config, [id]);
      if (!email) continue;
      const from = extractFromAddress(email);
      const subject = email.subject ?? '';
      if (!passesPreFilter(rules, { from, subject })) {
        markEmailProcessed(db, { emailId: id, filteredOut: true });
        continue;
      }
      if (llmCallsInLastDay(db) >= config.deepseek.dailyCap) break;
      const body = extractPlainText(email);
      const t0 = Date.now();
      try {
        const detection = await callDeepSeek({ config, from, subject, body });
        logLlmCall(db, { emailId: id, durationMs: Date.now() - t0, ok: true });
        markEmailProcessed(db, {
          emailId: id,
          filteredOut: false,
          isSubscription: detection.is_subscription_payment,
          detectionJson: JSON.stringify(detection)
        });
        if (detection.is_subscription_payment && detection.confidence >= 0.6) {
          detections.push({ email_id: id, raw_subject: subject, ...detection });
        }
      } catch (e) {
        logLlmCall(db, { emailId: id, durationMs: Date.now() - t0, ok: false });
        console.error('backfill deepseek error', id, e.message);
      }
    }
    res.json({ ok: true, count: detections.length, detections });
  });
  return r;
}

// SubhookAuth.kt
package com.subtracker.detect

import android.util.Base64
import javax.crypto.Mac
import javax.crypto.spec.SecretKeySpec

object SubhookAuth {
    fun sign(secret: String, tsMillis: Long, body: String): String {
        val mac = Mac.getInstance("HmacSHA256")
        mac.init(SecretKeySpec(secret.toByteArray(Charsets.UTF_8), "HmacSHA256"))
        val raw = "$tsMillis.$body".toByteArray(Charsets.UTF_8)
        val sig = Base64.encodeToString(mac.doFinal(raw), Base64.NO_WRAP)
        return "HMAC-SHA256 ts=$tsMillis,sig=$sig"
    }
}

// DetectionReconciler.kt
package com.subtracker.detect

import com.subtracker.PendingDetection
import com.subtracker.Subscription
import java.util.Locale

object DetectionReconciler {
    private const val AMOUNT_EPSILON = 0.01

    fun reconcile(
        payload: DetectionPayload,
        subscriptions: List<Subscription>,
        now: Long
    ): PendingDetection {
        val pName = payload.provider.lowercase(Locale.US)
        val match = subscriptions.firstOrNull {
            val n = it.name.lowercase(Locale.US)
            n.contains(pName) || pName.contains(n.substringBefore(' '))
        }
        val kind = when {
            match == null -> "new_sub"
            kotlin.math.abs(match.amount - payload.amount) > AMOUNT_EPSILON -> "amount_change"
            match.currency.uppercase(Locale.US) != payload.currency.uppercase(Locale.US) -> "amount_change"
            else -> "new_payment"
        }
        return PendingDetection(
            emailId = payload.email_id,
            kind = kind,
            targetSubId = match?.id,
            provider = payload.provider,
            amount = payload.amount,
            currency = payload.currency,
            dateIso = payload.date_iso,
            cycle = payload.cycle,
            confidence = payload.confidence,
            rawSubject = payload.raw_subject,
            createdAt = now,
            status = "pending"
        )
    }
}

// SubhookClient.kt
package com.subtracker.detect

import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext
import kotlinx.serialization.Serializable
import kotlinx.serialization.builtins.ListSerializer
import kotlinx.serialization.json.Json
import okhttp3.MediaType.Companion.toMediaType
import okhttp3.OkHttpClient
import okhttp3.Request
import okhttp3.RequestBody.Companion.toRequestBody
import java.util.concurrent.TimeUnit

private val json = Json { ignoreUnknownKeys = true; encodeDefaults = true }
private val mediaJson = "application/json".toMediaType()

@Serializable
private data class RegisterRequest(val token: String, val device_name: String? = null)

@Serializable
data class BackfillRequest(val months: Int)

@Serializable
data class BackfillResponse(val ok: Boolean, val count: Int = 0, val detections: List<DetectionPayload> = emptyList())

class SubhookClient(
    private val baseUrl: String,
    private val hmacSecret: String
) {
    private val client = OkHttpClient.Builder()
        .connectTimeout(15, TimeUnit.SECONDS)
        .readTimeout(120, TimeUnit.SECONDS)
        .build()

    suspend fun registerFcmToken(token: String, deviceName: String?) = withContext(Dispatchers.IO) {
        val body = json.encodeToString(RegisterRequest.serializer(), RegisterRequest(token, deviceName))
        val req = signedRequest("/register-fcm-token", body)
        client.newCall(req).execute().use { resp ->
            if (!resp.isSuccessful) error("register failed: ${resp.code}")
        }
    }

    suspend fun backfill(months: Int): BackfillResponse = withContext(Dispatchers.IO) {
        val body = json.encodeToString(BackfillRequest.serializer(), BackfillRequest(months))
        val req = signedRequest("/backfill", body)
        client.newCall(req).execute().use { resp ->
            val txt = resp.body?.string() ?: error("empty response")
            if (!resp.isSuccessful) error("backfill failed: ${resp.code} $txt")
            json.decodeFromString(BackfillResponse.serializer(), txt)
        }
    }

    private fun signedRequest(path: String, body: String): Request {
        val ts = System.currentTimeMillis()
        val auth = SubhookAuth.sign(hmacSecret, ts, body)
        return Request.Builder()
            .url(baseUrl.trimEnd('/') + path)
            .post(body.toRequestBody(mediaJson))
            .header("Authorization", auth)
            .build()
    }
}

class MailDetectionWorker(

) : CoroutineWorker(ctx, params) {
    override suspend fun doWork(): Result =
        try {
            val payload = DetectionPayload(
                email_id = inputData.getString("email_id") ?: return Result.failure(),
                provider = inputData.getString("provider").orEmpty(),
                amount = inputData.getString("amount")?.toDoubleOrNull() ?: 0.0,
                currency = inputData.getString("currency").orEmpty(),
                date_iso = inputData.getString("date_iso").orEmpty(),
                cycle = inputData.getString("cycle").orEmpty(),
                confidence = inputData.getString("confidence")?.toDoubleOrNull() ?: 0.0,
                raw_subject = inputData.getString("raw_subject").orEmpty()
            )
            val app = ctx.applicationContext as App
            val pendingDao = app.db.pendingDao()

            val subs = app.db.dao().getAll().first()
            val pd = DetectionReconciler.reconcile(payload, subs, System.currentTimeMillis())
            val newId = pendingDao.insert(pd)

                dao.update(sub.copy(amount = pd.amount, currency = pd.currency))
                paymentDao.insert(PaymentLog(
                    subscriptionId = sid, paidAt = parseIsoDateMillis(pd.dateIso),
                    amount = pd.amount, currency = pd.currency, cycleAtPayment = pd.cycle.ifEmpty { "monthly" }
                ))
            }
            "date_correction" -> pd.targetSubId?.let { sid ->
                val sub = dao.byId(sid) ?: return@let
                dao.update(sub.copy(nextBilling = parseIsoDateMillis(pd.dateIso)))
            }
        }
        db.pendingDao().setStatus(pd.id, "accepted")
    }
}

fun rejectDetection(pd: PendingDetection) = viewModelScope.launch {
    db.pendingDao().setStatus(pd.id, "rejected")
}

fun triggerBackfill(months: Int = 12): Flow<BackfillResult> = flow {
    emit(BackfillResult.Loading)
    try {
        val client = SubhookClient(BuildConfig.SUBHOOK_BASE_URL, BuildConfig.SUBHOOK_HMAC_SECRET)
        val resp = client.backfill(months)
        val subs = db.dao().getAll().first()
        val now = System.currentTimeMillis()
        var inserted = 0
        db.withTransaction {
            for (p in resp.detections) {
                if (db.pendingDao().byEmailId(p.email_id) != null) continue
                val pd = DetectionReconciler.reconcile(p, subs, now)
                if (db.pendingDao().insert(pd) > 0) inserted++
            }
        }
        emit(BackfillResult.Done(inserted))
    } catch (t: Throwable) {
        emit(BackfillResult.Error(t.message ?: "unknown"))
    }
}
```

internal fun parseIsoDateMillis(iso: String): Long {
    val d = java.time.LocalDate.parse(iso)
    return d.atStartOfDay(java.time.ZoneId.systemDefault()).toInstant().toEpochMilli()
}

    data class Done(val inserted: Int) : BackfillResult()
    data class Error(val message: String) : BackfillResult()

package com.subtracker.ui

import androidx.compose.foundation.background
import androidx.compose.foundation.layout.*

import com.subtracker.BackfillResult
import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.collect

@Composable
fun BackfillScreen(
    triggerBackfill: (Int) -> Flow<BackfillResult>,
    onBack: () -> Unit
) {
    var months by remember { mutableStateOf("12") }
    var state by remember { mutableStateOf<BackfillResult?>(null) }
    val scope = rememberCoroutineScope()

    Column(Modifier.fillMaxSize().background(Color(0xFFF4F3EF))) {
        Row(Modifier.fillMaxWidth().padding(start = 4.dp, end = 12.dp, top = 8.dp),
            verticalAlignment = Alignment.CenterVertically) {
            IconButton(onClick = onBack) { Icon(Icons.AutoMirrored.Filled.ArrowBack, "Geri") }
            Spacer(Modifier.width(4.dp))
            Text("Geçmiş Tarama", fontWeight = FontWeight.Bold, fontSize = 20.sp)
        }

        Column(Modifier.fillMaxWidth().padding(24.dp), verticalArrangement = Arrangement.spacedBy(16.dp)) {
            Text(
                "Fastmail INBOX'unu tarayıp eksik abonelik ve ödemeleri tespit eder. " +
                "Her tespit bekleyen onaya düşer.",
                fontSize = 13.sp, color = Color(0xFF77736C)
            )
            OutlinedTextField(
                value = months,
                onValueChange = { if (it.all { c -> c.isDigit() }) months = it },
                label = { Text("Kaç ay geriye git?") },
                singleLine = true,
                modifier = Modifier.fillMaxWidth()
            )
            Button(
                onClick = {
                    val n = months.toIntOrNull()?.coerceIn(1, 24) ?: 12
                    scope.launch {
                        triggerBackfill(n).collect { state = it }
                    }
                },
                enabled = state !is BackfillResult.Loading,
                modifier = Modifier.fillMaxWidth().height(52.dp),
                shape = RoundedCornerShape(26.dp),
                colors = ButtonDefaults.buttonColors(containerColor = Color(0xFF111111))
            ) {
                Text(if (state is BackfillResult.Loading) "Taranıyor..." else "Başlat",
                    fontWeight = FontWeight.SemiBold)
            }
            when (val s = state) {
                is BackfillResult.Loading -> CircularProgressIndicator()
                is BackfillResult.Done -> Text("${s.inserted} tespit eklendi — Mail Tespitleri ekranına bak.",
                    color = Color(0xFF4CAF50), fontWeight = FontWeight.SemiBold)
                is BackfillResult.Error -> Text("Hata: ${s.message}", color = Color(0xFFD32F2F))
                null -> {}
            }
        }
    }
}
