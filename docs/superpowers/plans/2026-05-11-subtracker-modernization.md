> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Mevcut SubTracker görsel kimliğini koruyarak bildirim + ödeme geçmişi feature'larını ekle, ölü kodu ve güvenlik açıklarını temizle.

**Architecture:** Mevcut Compose/Room iskeleti üzerine yeni Room entity (`PaymentLog`), DB migration 1→2, WorkManager-based periodic reminder worker, ve 2 yeni screen (Notifications, PaymentHistory). Navigation sealed class state machine'e geçer. UI değişiklikleri sadece additive — mevcut renkler/kart stilleri dokunulmuyor.

**Tech Stack:** Kotlin 1.9.24, Android compileSdk 34 minSdk 26, Jetpack Compose BOM 2024.02.00, Material3, Room 2.6.1, WorkManager 2.9.0, KSP, JUnit4 + Room testing + Work testing.

```kotlin
package com.subtracker

import android.content.Context
import android.util.Xml
import androidx.test.core.app.ApplicationProvider
import kotlinx.coroutines.test.runTest
import org.junit.Assert.assertEquals
import org.junit.Assert.assertNull
import org.junit.Assert.assertTrue
import org.junit.Test
import org.junit.runner.RunWith
import org.robolectric.RobolectricTestRunner
import org.xmlpull.v1.XmlPullParser
import java.io.InputStream

@RunWith(RobolectricTestRunner::class)
class ExchangeRateRepositoryTest {

    private val ctx: Context get() = ApplicationProvider.getApplicationContext()

    @Test fun `parseTcmbXml extracts ForexSelling rates`() {
        val stream: InputStream = javaClass.classLoader!!
            .getResourceAsStream("tcmb_today_sample.xml")!!
        val parser = Xml.newPullParser().apply { setInput(stream, "UTF-8") }
        val rates = mutableMapOf("TRY" to 1.0)
        var currentCode: String? = null
        var currentTag: String? = null
        while (parser.next() != XmlPullParser.END_DOCUMENT) {
            when (parser.eventType) {
                XmlPullParser.START_TAG -> {
                    currentTag = parser.name
                    if (parser.name == "Currency") currentCode = parser.getAttributeValue(null, "Kod")
                }
                XmlPullParser.TEXT -> {
                    if (currentTag == "ForexSelling") {
                        val v = parser.text.trim().replace(',', '.').toDoubleOrNull()
                        if (currentCode != null && v != null) rates[currentCode!!.uppercase()] = v
                    }
                }
                XmlPullParser.END_TAG -> {
                    if (parser.name == "Currency") currentCode = null
                    currentTag = null
                }
            }
        }
        assertEquals(45.27, rates["USD"]!!, 0.0001)
        assertEquals(48.96, rates["EUR"]!!, 0.0001)
        assertEquals(1.0, rates["TRY"]!!, 0.0001)
    }

    @Test fun `loadCached returns null when prefs empty`() = runTest {
        ctx.getSharedPreferences("exchange_rates", Context.MODE_PRIVATE).edit().clear().apply()
        val repo = ExchangeRateRepository(ctx)
        val method = repo.javaClass.getDeclaredMethod("loadCached")
        method.isAccessible = true
        val result = method.invoke(repo)
        assertNull(result)
    }

    @Test fun `save then loadCached round-trips rates`() = runTest {
        val prefs = ctx.getSharedPreferences("exchange_rates", Context.MODE_PRIVATE)
        prefs.edit().clear().apply()
        val repo = ExchangeRateRepository(ctx)

        val sample = ExchangeRates(
            ratesToTry = mapOf("TRY" to 1.0, "USD" to 45.27),
            sourceDate = "11.05.2026",
            source = "TCMB satış"
        )

        val saveMethod = repo.javaClass.getDeclaredMethod("save", ExchangeRates::class.java)
        saveMethod.isAccessible = true
        saveMethod.invoke(repo, sample)

        val loadMethod = repo.javaClass.getDeclaredMethod("loadCached")
        loadMethod.isAccessible = true
        val loaded = loadMethod.invoke(repo) as ExchangeRates

        assertEquals(45.27, loaded.ratesToTry["USD"]!!, 0.0001)
        assertEquals("11.05.2026", loaded.sourceDate)
        assertTrue(loaded.source.contains("TCMB"))
    }
}
```

import android.app.NotificationChannel
import android.app.NotificationManager
import android.app.PendingIntent
import android.content.Context
import android.content.Intent
import android.os.Build
import androidx.core.app.NotificationCompat
import androidx.core.content.ContextCompat

import androidx.work.ExistingPeriodicWorkPolicy
import androidx.work.PeriodicWorkRequestBuilder
import androidx.work.WorkManager
import androidx.work.WorkerParameters
import java.util.concurrent.TimeUnit

object Notifier {
    const val CHANNEL_ID = "subscription_reminders"
    const val WORK_NAME = "reminder_check"

    // Test seams
    var dbOverrideForTest: AppDb? = null
    var notifyHookForTest: ((Subscription) -> Unit)? = null

    fun ensureChannel(ctx: Context) {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
            val nm = ContextCompat.getSystemService(ctx, NotificationManager::class.java) ?: return
            val ch = NotificationChannel(
                CHANNEL_ID,
                "Abonelik hatırlatıcıları",
                NotificationManager.IMPORTANCE_DEFAULT
            ).apply {
                description = "Yaklaşan abonelik ödemeleri için bildirim"
            }
            nm.createNotificationChannel(ch)
        }
    }

    fun schedulePeriodic(ctx: Context) {
        val request = PeriodicWorkRequestBuilder<ReminderWorker>(24, TimeUnit.HOURS)
            .setInitialDelay(1, TimeUnit.HOURS)
            .build()
        WorkManager.getInstance(ctx).enqueueUniquePeriodicWork(
            WORK_NAME,
            ExistingPeriodicWorkPolicy.KEEP,
            request
        )
    }

    fun fire(ctx: Context, sub: Subscription) {
        notifyHookForTest?.invoke(sub)?.also { return }

        val intent = Intent(ctx, MainActivity::class.java).apply {
            flags = Intent.FLAG_ACTIVITY_CLEAR_TOP or Intent.FLAG_ACTIVITY_SINGLE_TOP
        }
        val pi = PendingIntent.getActivity(
            ctx, sub.id.toInt(), intent,
            PendingIntent.FLAG_UPDATE_CURRENT or PendingIntent.FLAG_IMMUTABLE
        )

        val daysOut = ((sub.nextBilling - System.currentTimeMillis()) / 86_400_000L)
            .coerceAtLeast(0)
        val whenText = if (daysOut <= 0) "bugün" else "$daysOut gün sonra"

        val notification = NotificationCompat.Builder(ctx, CHANNEL_ID)
            .setSmallIcon(android.R.drawable.ic_dialog_info)
            .setContentTitle("${sub.name} ödemesi")
            .setContentText("${formatMoney(sub.amount, sub.currency)} $whenText düşecek")
            .setContentIntent(pi)
            .setAutoCancel(true)
            .setPriority(NotificationCompat.PRIORITY_DEFAULT)
            .build()

        val nm = ContextCompat.getSystemService(ctx, NotificationManager::class.java) ?: return
        nm.notify(sub.id.toInt(), notification)
    }
}

class ReminderWorker(
    private val ctx: Context,
    params: WorkerParameters,
    private val nowProvider: () -> Long = { System.currentTimeMillis() }
) : CoroutineWorker(ctx, params) {

    override suspend fun doWork(): Result {
        val db = Notifier.dbOverrideForTest ?: (ctx.applicationContext as App).db
        val now = nowProvider()
        val subs = db.dao().all()
        subs.filter { it.reminderOn }
            .filter { sub ->
                val delta = sub.nextBilling - now
                delta in 0..(sub.reminderDays.toLong() * 86_400_000L)
            }
            .forEach { Notifier.fire(ctx, it) }
        return Result.success()
    }
}

@Composable
fun PaymentHistoryScreen(

    subscriptions: List<Subscription>,
    onBack: () -> Unit
) {
    val zoneId = ZoneId.systemDefault()
    val grouped: Map<YearMonth, List<PaymentLog>> = payments
        .groupBy { YearMonth.from(Instant.ofEpochMilli(it.paidAt).atZone(zoneId).toLocalDate()) }

    Column(Modifier.fillMaxSize()) {
        Row(
            Modifier.fillMaxWidth().padding(start = 4.dp, end = 12.dp, top = 8.dp),
            verticalAlignment = Alignment.CenterVertically
        ) {

            Text("Ödeme geçmişi", fontWeight = FontWeight.Bold, fontSize = 20.sp)
        }

        if (payments.isEmpty()) {
            Column(
                Modifier.fillMaxSize(),
                verticalArrangement = Arrangement.Center,
                horizontalAlignment = Alignment.CenterHorizontally
            ) {
                Text("Henüz ödeme yok", fontWeight = FontWeight.Bold, fontSize = 18.sp)
                Spacer(Modifier.height(6.dp))
                Text(
                    "Abonelikler vade geldikçe burada listelenir.",
                    color = Color(0xFF77736C), fontSize = 13.sp
                )
            }
            return@Column
        }

        LazyColumn(
            Modifier.fillMaxSize().padding(horizontal = 16.dp),
            verticalArrangement = Arrangement.spacedBy(10.dp),
            contentPadding = PaddingValues(top = 8.dp, bottom = 24.dp)
        ) {
            grouped.toSortedMap(compareByDescending { it }).forEach { (ym, logs) ->
                val monthTotal = logs.sumOf { it.amount }
                val firstCurrency = logs.first().currency
                item {
                    Row(
                        Modifier.fillMaxWidth()
                            .background(Color(0xFFF4F3EF))
                            .padding(vertical = 8.dp, horizontal = 4.dp),
                        horizontalArrangement = Arrangement.SpaceBetween
                    ) {
                        val label = "${ym.month.getDisplayName(TextStyle.FULL, Locale("tr","TR"))} ${ym.year}"
                        Text(label, fontWeight = FontWeight.Black, fontSize = 13.sp, color = Color(0xFF9A968F))
                        Text(
                            formatMoney(monthTotal, firstCurrency),
                            fontWeight = FontWeight.Bold,
                            fontSize = 13.sp,
                            color = Color(0xFF77736C)
                        )
                    }
                }
                items(logs) { log ->
                    val sub = subscriptions.find { it.id == log.subscriptionId }
                    PaymentLogRow(log, sub)
                }
            }
        }
    }
}
